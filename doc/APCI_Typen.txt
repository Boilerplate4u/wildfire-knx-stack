apciGROUP_VALUE_READ					Read object value 			0000 	0x00	no Data
apciGROUP_VALUE_RESP					Response object value		0001 	0x01	value of the object (1 bit - 14 byte)
apciGROUP_VALUE_WRITE					Write object value			0010 	0x02	value of the object (1 bit - 14 byte)

apciINDIVIDUAL_ADDRESS_WRITE	Write phys. address			0011 	0x03	physical address (2 byte)
apciINDIVIDUAL_ADDRESS_READ		Read phys. address			0100 	0x04	no data
apciINDIVIDUAL_ADDRESS_RESP		Response phys. address	0101 	0x05	no data (PA is contained in the source address)

apciADC_READ									Read AD-converter				0110 	0x06	AD-port-number (6 bit), number of repetitions (1 byte)
apciADC_RESP									Response AD-value				0111 	0x07	AD-port-number (6 bit), number of repetitions (1 byte), AD-converter-value (2 byte)
apciMEMORY_READ								Read memory							1000 	0x08	number of bytes (4 bit), offset address (2 byte)
apciMEMORY_RESP								Response memory					1001 	0x09	number of bytes (4 bit), offset address (2 byte), memory data (1 byte - 12 byte)
apciMEMORY_WRITE							Write Memory						1010 	0x0A	number of bytes (4 bit), offset address (2 byte), memory data (1 byte - 12 byte)

apciUSER_MSG									Message to user					1011 	0x0B	message (at the most 14 bytes + 6 bit)

apciDEVICE_DESCRIPTOR_READ		Read BAU-maskversion		1100 	0x0C	no data
apciDEVICE_DESCRIPTOR_RESP		Response BAU-maskver.		1101 	0x0D	mask type byte (1 byte), version byte (1 byte)

apciRESTART										Restart									1110 	0x0E	no data

apciESCAPE										<escape> 								1111 	0x0F	used for management services
